# %%
def show_res(fn):
  im = np.array(PILImage.create(fn))
  out = learn.predict(fn)[1]
  mask = np.array(Image.fromarray(out.numpy().astype(np.uint8)).resize((1920,1088)))
  im[mask==1, 0] = 255
  im[mask==2, 1] = 255
  f, axs = plt.subplots(1,1, figsize=(16,16))
  axs.imshow(im)
